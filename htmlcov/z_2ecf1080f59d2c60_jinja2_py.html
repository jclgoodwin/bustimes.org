25 def parse(self, parser): 

26 # the first token is the token that started the tag. In our case 

27 # we only listen to ``'cache'`` so this will be a name token with 

28 # `cache` as value. We get the line number so that we can give 

29 # that line number to the nodes we create by hand. 

30 lineno = next(parser.stream).lineno 

31 

32 # now we parse a single expression that is used as cache key. 

33 args = [parser.parse_expression()] 

34 

35 # if there is a comma, the user provided a timeout. If not use 

36 # None as second parameter. 

37 if parser.stream.skip_if("comma"): 

38 args.append(parser.parse_expression()) 

39 else: 

40 args.append(nodes.Const(None)) 

41 

42 # now we parse the body of the cache block up to `endcache` and 

43 # drop the needle (which would always be `endcache` in that case) 

44 body = parser.parse_statements(["name:endcache"], drop_needle=True) 

45 

46 # now return a `CallBlock` node that calls our _cache_support 

47 # helper method on this extension. 

48 return nodes.CallBlock( 

49 self.call_method("_cache_support", args), [], [], body 

50 ).set_lineno(lineno) 

51 

52 def _cache_support(self, name, timeout, caller): 

53 """Helper callback.""" 

54 key = self.environment.fragment_cache_prefix + name 

55 

56 # try to load the block from the cache 

57 # if there is no fragment in the cache, render it and store 

58 # it in the cache. 

59 rv = cache.get(key) 

60 if rv is not None: 

61 return rv 

62 rv = caller() 

63 cache.set(key, rv, timeout) 

64 return rv
